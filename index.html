// Template set
setTpl: function( tpl ){
    var regExp = /\{\s*(\S*?)\s*}/g;
    $.tpl = tpl.replace( regExp, '{$1}');
},

// Get var RegExp
getRegExp: function( key ){
    var str = '{\\$' + key + '}';
    var reg = new RegExp( str, "gim" );
    return reg;
},

// Assign value
assign: function( key, value ){
    $.valueList[ key ] = value;
},
